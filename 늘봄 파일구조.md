# 늘봄 파일구조

neulbom-project/
├── neulbom-frontend/            # [Frontend] React + Vite
│   ├── public/
│   │   ├── manifest.json        # PWA 설정 (앱 이름, 아이콘)
│   │   ├── service-worker.js    # PWA 오프라인 지원 스크립트
│   │   └── icons/               # 홈 화면용 아이콘 이미지
│   ├── src/
│   │   ├── api/                 # Axios API 호출 함수 (auth, chat, welfare)
│   │   ├── assets/              # 정적 이미지, 폰트
│   │   ├── components/          # 재사용 UI (Navbar, ChatBubble, WelfareCard)
│   │   ├── pages/               # 라우팅 페이지
│   │   │   ├── Home.jsx         # 메인 (챗봇 바로가기)
│   │   │   ├── Chat.jsx         # AI 상담 채팅방
│   │   │   ├── Welfare.jsx      # 복지 정보 검색/결과
│   │   │   ├── Community.jsx    # 익명 게시판
│   │   │   └── Login.jsx
│   │   ├── App.jsx              # 라우터 설정
│   │   └── main.jsx             # 진입점
│   ├── index.html               # 웹 엔트리 포인트
│   ├── vite.config.js           # 빌드 설정
│   └── package.json
│
└── neulbom-backend/             # [Backend] FastAPI
├── app/
│   ├── main.py                  # 앱 실행 진입점
│   ├── database.py              # SQLite DB 연결 설정
│   │   ├── __init__.py
│   ├── models.py                # DB 테이블 정의 (User, Welfare, Post)
│   │   ├── connection.py        # (구 database.py) SQLite 연결 설정
│   │   ├── models.py            # 테이블 정의 (User, Post 등)
│   │   ├── crud.py              # 데이터 조회/저장 함수 모음
│   │   └── schema.py            # (구 schemas.py) 데이터 검증용 DTO
│   │
│   ├── api/                     # [API 진입점] 클라이언트 요청을 받는 곳
│   │   ├── endpoints/
│   │   │   ├── auth.py          # 로그인/회원가입
│   │   │   ├── chat.py          # "채팅 메시지 보내기" 요청 처리
│   │   │   ├── welfare.py       # "복지 정보 검색해줘" 요청 처리
│   │   │   └── community.py     # "글 쓸래/심사해줘" 요청 처리
│   │
│   ├── ai_core/                 # [AI의 두뇌] LLM, RAG, 프롬프트 관리 (핵심!)
│   │   ├── __init__.py
│   │   ├── llm_client.py        # OpenAI/Upstage API 호출 래퍼 (Wrapper)
│   │   ├── prompts.py           # [중요] "너는 상담사야", "17세 처럼 요약해" 프롬프트 모음
│   │   ├── rag_engine.py        # 벡터 검색(FAISS/Chroma) 및 문서 검색 로직
│   │   ├── safety_guard.py      # [위기 감지] 자살/학대 키워드 필터링 로직
│   │   └── text_processor.py    # PDF 텍스트 추출 및 전처리 (Chunking)
│   │
│   ├── services/                # [비즈니스 로직] AI 결과와 DB를 엮는 곳
│   │   ├── chat_service.py      # 대화 내역 저장 + 위기 감지 결과에 따른 분기 처리
│   │   ├── welfare_service.py   # 사용자 프로필(나이/지역) + RAG 검색 결과 매칭
│   │   ├── community_service.py # 게시글 AI 클린봇 검수 후 DB 저장 로직
│   │   └── auth_service.py      # JWT 토큰 발급 및 인증
│   │
│   └── core/                    # 설정 파일
│       ├── config.py            # 환경변수 로드 (.env)
│       └── exceptions.py        # 에러 처리
│
├── data/                        # RAG용 데이터 저장소
│   ├── raw_pdfs/                # 크롤링한 복지 공고문 PDF 파일들
│   └── vector_db/               # 임베딩된 벡터 데이터 저장 폴더 (FAISS 인덱스)
│
├── neulbom.db                   # SQLite 데이터베이스 (User, ChatLog 등)
└── requirements.txt
